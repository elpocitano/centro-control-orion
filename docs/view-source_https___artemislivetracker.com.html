════════════════════════════════════════════════ */
var EX=0.26,EY=0.52; // Earth normalised
var MX=0.84,MY=0.38; // Moon normalised

function bez(p0,p1,p2,p3,t){
var m=1-t;
return{
x:m*m*m*p0.x+3*m*m*t*p1.x+3*m*t*t*p2.x+t*t*t*p3.x,
y:m*m*m*p0.y+3*m*m*t*p1.y+3*m*t*t*p2.y+t*t*t*p3.y
};
}

function traj(t){
t=Math.max(0,Math.min(t,MH));
if(t<0.5){
// Ascent — straight up from KSC (right side Earth)
var f=t/0.5;
return{x:EX+0.035+f*0.01,y:EY-0.032-f*0.055};
}
if(t<2){
// Initial orbit arc
var f=(t-0.5)/1.5, a=-Math.PI/2+f*Math.PI*0.62;
return{x:EX+Math.cos(a)*0.072,y:EY+Math.sin(a)*0.052};
}
if(t<5){
// Prox Ops — completing first orbit loops
var f=(t-2)/3, a=-Math.PI/2+Math.PI*0.62+f*Math.PI*0.75;
return{x:EX+Math.cos(a)*0.092,y:EY+Math.sin(a)*0.062};
}
if(t<24){
// HEO — larger elliptical orbit (multiple loops)
var f=(t-5)/19, a=-Math.PI/2+Math.PI*1.37+f*Math.PI*2.15;
return{x:EX+Math.cos(a)*0.115,y:EY+Math.sin(a)*0.078};
}
if(t<27){
// TLI departure — spiraling outward
var f=(t-24)/3, base=-Math.PI/2+Math.PI*1.37+Math.PI*2.15;
var a=base+f*0.75, r=0.115+f*0.075;
return{x:EX+Math.cos(a)*r,y:EY+Math.sin(a)*r*0.70};
}
if(t<118){
// Outbound translunar coast — upper arc
var f=(t-27)/91;
var p0={x:EX+0.14,y:EY-0.10};
var p1={x:EX+0.20,y:EY-0.30}; // peak (ICPS disposal region top)
var p2={x:EX+0.44,y:EY-0.26}; // sweeping right
var p3={x:MX+0.04,y:MY-0.08}; // Moon approach
return bez(p0,p1,p2,p3,f);
}
if(t<135){
// Lunar flyby — arc around far side
var f=(t-118)/17;
var p0={x:MX+0.04,y:MY-0.08};
var p1={x:MX+0.13,y:MY+0.02};
var p2={x:MX+0.09,y:MY+0.13};
var p3={x:MX-0.06,y:MY+0.10};
return bez(p0,p1,p2,p3,f);
}
if(t<145){
// Post-flyby transition
var f=(t-135)/10;
var p0={x:MX-0.06,y:MY+0.10};
var p1={x:MX-0.11,y:MY+0.19};
var p2={x:MX-0.21,y:MY+0.22};
var p3={x:MX-0.30,y:MY+0.20};
return bez(p0,p1,p2,p3,f);
}
// Return lower arc back to Earth
var f=(t-145)/(MH-145);
var p0={x:MX-0.30,y:MY+0.20};
var p1={x:EX+0.36,y:EY+0.36};
var p2={x:EX+0.17,y:EY+0.27};
var p3={x:EX+0.032,y:EY+0.048};
return bez(p0,p1,p2,p3,f);
}

function wp(p){return{x:p.x*W,y:p.y*H};}

// Pre-build path points
var PP=[];
function buildPath(){
PP=[];
for(var t=0;t<=MH;t+=0.35){
var p=traj(t);PP.push({x:p.x*W,y:p.y*H,t:t});
}
}

LOCATE CRAFT — Earth-Moon POV

function locate(animate){
var met=getMET(), t=Math.min(Math.max(0,met),MH);
var np=traj(t), ox=np.x*W, oy=np.y*H;
var ex=EX*W,ey=EY*H,mx2=MX*W,my2=MY*H;
var pad=0.28;
var minX=Math.min(ex,ox,mx2)-20, maxX=Math.max(ex,ox,mx2)+20;
var minY=Math.min(ey,oy,my2)-20, maxY=Math.max(ey,oy,my2)+20;
var bW=Math.max(maxX-minX,100), bH=Math.max(maxY-minY,80);
var ts=Math.min(W/(bW*(1+pad*2)),H/(bH*(1+pad*2)),3.8);
var midX=(minX+maxX)/2,midY=(minY+maxY)/2;
var tvx=W/2-midX*ts, tvy=H/2-midY*ts;
if(!animate){vx=tvx;vy=tvy;vs=ts;dirty=true;return;}
anim={vx:tvx,vy:tvy,vs:ts};
}

function animView(){
if(!anim)return;
var sp=0.1;
vx+=(anim.vx-vx)*sp;vy+=(anim.vy-vy)*sp;vs+=(anim.vs-vs)*sp;dirty=true;
if(Math.abs(vx-anim.vx)<0.4&&Math.abs(vy-anim.vy)<0.4&&Math.abs(vs-anim.vs)<0.04){
vx=anim.vx;vy=anim.vy;vs=anim.vs;anim=null;
}
}

function arowDecode(json){
try{
var d=JSON.parse(json);
if(!d.Parameter_2003){AROW_T=Date.now()-43000;return null;} // sparse: retry in 15s
var x=parseFloat(d.Parameter_2003.Value);
var y=parseFloat(d.Parameter_2004.Value);
var z=parseFloat(d.Parameter_2005.Value);
var vx=parseFloat(d.Parameter_2009.Value);
var vy=parseFloat(d.Parameter_2010.Value);
var vz=parseFloat(d.Parameter_2011.Value);
var ts=parseFloat(d.Parameter_5010.Value)*1000;
if(isNaN(x)||isNaN(vx)||isNaN(ts))return null;
var dE_c=Math.sqrt(x*x+y*y+z*z)*0.0003048; // feet→km from center
var dE=dE_c-6371; // km from surface
var speed=Math.sqrt(vx*vx+vy*vy+vz*vz)*0.0003048*3600; // ft/s→km/h
var sig=(dE_c/299792.458).toFixed(3);
var launchMs=Date.UTC(2026,3,1,22,35,0);
var metH=(ts-launchMs)/3600000;
var md=384400-Math.sin(metH/240*Math.PI)*18000;
var ang=metH*0.549*Math.PI/180+2.8;
var mx=md/0.0003048*Math.cos(ang),my=md/0.0003048*Math.sin(ang),mz=md/0.0003048*0.09;
var dM=Math.sqrt(Math.pow(x-mx,2)+Math.pow(y-my,2)+Math.pow(z-mz,2))*0.0003048;
return{dE:Math.max(0,Math.round(dE)),dM:Math.round(dM),speed:Math.round(speed),
